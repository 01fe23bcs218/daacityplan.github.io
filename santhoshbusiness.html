            // Dijkstra's Algorithm for Waste Collection Optimization
            function dijkstra(graph, start) {
                let distances = {};
                let visited = {};
                let pq = new PriorityQueue();
                pq.enqueue(start, 0);
                distances[start] = 0;
                while (!pq.isEmpty()) {
                    let node = pq.dequeue();
                    visited[node] = true;
                    let neighbors = graph[node];
                    for (let neighbor in neighbors) {
                        if (!visited[neighbor]) {
                            let newDist = distances[node] + neighbors[neighbor];
                            if (newDist < (distances[neighbor] || Infinity)) {
                                distances[neighbor] = newDist;
                                pq.enqueue(neighbor, newDist);
                            }
                        }
                    }
                }
                return distances;
            }

Business Case 2: Intelligent Urban Transport System

SDG 11 Alignment

Target 11.2: Provide access to safe, affordable, and sustainable transport systems.

Indicator 11.2.1: Proportion of population with convenient access to public transport.

Proposed Algorithms

  • Breadth-First Search (BFS) for Public Transport Connectivity Analysis: Identify least-cost paths and connectivity across urban transport networks.
  • AVL Trees for Traffic Flow Management: Store and retrieve real-time traffic data efficiently.
  • Pruning for Congestion Detection: Remove less critical routes from evaluation to focus on high-impact traffic zones.

Business Case 2 Code

            // BFS for Public Transport Connectivity Analysis
            function bfs(graph, start) {
                let visited = new Set();
                let queue = [start];
                visited.add(start);
                while (queue.length > 0) {
                    let node = queue.shift();
                    let neighbors = graph[node];
                    for (let neighbor of neighbors) {
                        if (!visited.has(neighbor)) {
                            visited.add(neighbor);
                            queue.push(neighbor);
                        }
                    }
                }
                return visited;
            }

Business Case 3: Urban Green Space Planning

SDG 11 Alignment

Target 11.7: Provide universal access to safe, inclusive, and accessible green spaces.

Indicator 11.7.1: Average share of the built-up area that is open space for public use.

Proposed Algorithms

  • Backtracking for Optimal Park Placement: Explore multiple layouts to maximize green space while minimizing land use.
  • Segment Trees for Land Use Analysis: Track and query changes in land use over time efficiently.
  • Fenwick Trees for Green Space Accessibility Measurement: Measure accessibility by calculating distances between green spaces and residential areas.

Business Case 3 Code

            // Backtracking for Optimal Park Placement
            function backtrack(grid, landAvailable) {
                if (landAvailable <= 0) return true;
                for (let i = 0; i < grid.length; i++) {
                    for (let j = 0; j < grid[i].length; j++) {
                        if (grid[i][j] === 0) {
                            grid[i][j] = 1;
                            if (backtrack(grid, landAvailable - 1)) return true;
                            grid[i][j] = 0;
                        }
                    }
                }
                return false;
            }
